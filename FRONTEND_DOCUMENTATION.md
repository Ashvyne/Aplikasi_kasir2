```javascript
function pageSwitch(pageName) {
  // 1. Hide semua pages
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'))
  
  // 2. Remove active class dari menu
  document.querySelectorAll('.nav-item').forEach(item => 
    item.classList.remove('active')
  )
  
  // 3. Show selected page
  document.getElementById(pageName + '-page').classList.add('active')
  
  // 4. Highlight menu item
  document.querySelector(`[data-page="${pageName}"]`).classList.add('active')
  
  // 5. Update page title
  updatePageTitle(pageName)
  
  // 6. Load data untuk page
  if (pageName === 'products') loadProducts()
  else if (pageName === 'transactions') loadTransactions()
  else if (pageName === 'reports') loadReports()
}
```

#### 3. Add to Cart
```javascript
function addToCart(productId) {
  try {
    // Find product
    const product = products.find(p => p.id === productId)
    if (!product) return
    
    // Check jika sudah ada di cart
    const existingItem = cart.find(item => item.product_id === productId)
    
    if (existingItem) {
      // Update quantity
      if (existingItem.quantity < product.stock) {
        existingItem.quantity++
        existingItem.subtotal = existingItem.quantity * existingItem.price
      } else {
        showAlert('Stok tidak cukup')
        return
      }
    } else {
      // Add new item
      cart.push({
        product_id: productId,
        name: product.name,
        price: product.price,
        quantity: 1,
        subtotal: product.price,
        stock: product.stock
      })
    }
    
    // Update display
    displayCartItems()
    calculateTotals()
  } catch (error) {
    console.error('‚ùå Add to cart error:', error)
  }
}
```

#### 5. Calculate Totals
```javascript
function calculateTotals() {
  // Calculate subtotal
  let subtotal = 0
  cart.forEach(item => {
    subtotal += item.subtotal
  })
  
  // Get discount
  const discount = parseInt(document.getElementById('discountAmount').value || 0)
  
  // Calculate total
  const total = subtotal - discount
  
  // Update display
  document.getElementById('subtotal').textContent = formatPrice(subtotal)
  document.getElementById('discount').textContent = formatPrice(discount)
  document.getElementById('total').textContent = formatPrice(total)
  
  // Calculate change
  const cashReceived = parseInt(document.getElementById('cashReceived').value || 0)
  const change = cashReceived - total
  document.getElementById('changeAmount').textContent = formatPrice(change)
  
  // Warn jika not enough cash
  if (change < 0) {
    document.getElementById('changeAmount').style.color = 'red'
  } else {
    document.getElementById('changeAmount').style.color = 'green'
  }
}
```

#### 6. Create Transaction
```javascript
async function createTransaction() {
  try {
    // Validation
    if (cart.length === 0) {
      showAlert('Keranjang kosong!', 'warning')
      return
    }
    
    const total = parseInt(document.getElementById('total').textContent)
    const paymentMethod = document.getElementById('paymentMethod').value
    
    // Prepare data
    const transactionData = {
      items: cart,
      total: total,
      discount: parseInt(document.getElementById('discountAmount').value || 0),
      paymentMethod: paymentMethod
    }
    
    // Send to server
    const response = await fetch('/api/transactions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${localStorage.getItem('token')}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(transactionData)
    })
    
    const result = await response.json()
    
    if (response.ok) {
      // Show invoice
      showInvoice(result.invoiceNumber)
      
      // Clear cart
      cart = []
      displayCartItems()
      calculateTotals()
      
      // Refresh products (update stock)
      loadProducts()
      
      showAlert('Transaksi berhasil!', 'success')
    } else {
      showAlert(result.message || 'Gagal membuat transaksi', 'danger')
    }
  } catch (error) {
    console.error('‚ùå Create transaction error:', error)
    showAlert('Error: ' + error.message, 'danger')
  }
}
```

#### Save Product
```javascript
async function saveProduct() {
  try {
    const productId = document.getElementById('productId').value
    const formData = new FormData()
    
    // Collect form data
    formData.append('name', document.getElementById('productName').value)
    formData.append('sku', document.getElementById('productSku').value)
    formData.append('price', document.getElementById('productPrice').value)
    formData.append('stock', document.getElementById('productStock').value)
    
    // Handle image upload
    const imageFile = document.getElementById('productImage').files[0]
    if (imageFile) {
      formData.append('file', imageFile)
    }
    
    // Determine endpoint
    const method = productId ? 'PUT' : 'POST'
    const endpoint = productId ? `/api/products/${productId}` : '/api/products'
    
    // Send request
    const response = await fetch(endpoint, {
      method: method,
      headers: {
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      },
      body: formData
    })
    
    if (response.ok) {
      closeProductModal()
      loadProducts()
      showAlert('Product saved successfully!', 'success')
    } else {
      showAlert('Failed to save product', 'danger')
    }
  } catch (error) {
    console.error('‚ùå Save product error:', error)
  }
}
```

```javascript
function viewTransaction(transactionId) {
  try {
    // Find transaction
    const trans = transactions.find(t => t.id === transactionId)
    
    if (trans) {
      // Format data
      const dateObj = new Date(trans.createdAt)
      const date = dateObj.toLocaleDateString('id-ID', {
        weekday: 'long',
        year: 'numeric',
        month: 'long',
        day: 'numeric',
        hour: '2-digit',
        minute: '2-digit'
      })
      
      // Parse items
      let items = trans.items || []
      if (typeof items === 'string') {
        items = JSON.parse(items)
      }
      
      // Calculate subtotal
      let subtotal = 0
      items.forEach(item => {
        subtotal += (item.price || 0) * (item.quantity || 1)
      })
      
      // Show detail via modal
      Swal.fire({
        title: 'üìã Detail Transaksi',
        html: `
          <div class="transaction-detail">
            <div><strong>Invoice:</strong> ${trans.invoiceNumber}</div>
            <div><strong>Tanggal:</strong> ${date}</div>
            <div><strong>Items (${items.length}):</strong></div>
            ${items.map(item => `
              <div>${item.name} √ó ${item.quantity} = Rp ${formatPrice(item.price * item.quantity)}</div>
            `).join('')}
            <hr />
            <div><strong>Total:</strong> Rp ${formatPrice(trans.total)}</div>
          </div>
        `,
        icon: 'info',
        confirmButtonText: '‚úì Tutup'
      })
    }
  } catch (error) {
    console.error('‚ùå View transaction error:', error)
  }
}
```

### Utility Functions

#### Format Price (Rp Currency)
```javascript
function formatPrice(number) {
  if (!number) return '0'
  
  return new Intl.NumberFormat('id-ID', {
    style: 'currency',
    currency: 'IDR',
    minimumFractionDigits: 0,
    maximumFractionDigits: 0
  }).format(number)
}
```

#### Escape HTML (XSS Prevention)
```javascript
function escapeHtml(unsafe) {
  return unsafe
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#039;')
}
```
